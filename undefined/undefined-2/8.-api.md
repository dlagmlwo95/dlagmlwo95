## 3. Looping over an array

```javascript
for (let i = 0; i < fruits.length; i++){
     console.log(fruits[i]);
}; 

//for of
for(let fruit of fruits){
     console.log(fruit);
};

//forEach(콜백함수)
fruits.forEach(function(fruits,index){
    console.log(fruits,index);
}); 
//
fruits.forEach((fruit) => console.log(fruit));
```

## 4.Addtion, deletion, copy

```javascript
const fruits = ['사과', '바나나']
//push:넣기
fruits.push('딸기','복숭아');
console.log(fruits); //사과 ㅏ나나 딸기 

//pop :빼기
fruits.pop(); //사과 바나나 딸기
fruits.pop(); //사과 

// unshift 
fruits.unshift('레몬','수박'); //레몬 수박 사과 

//shift
fruits.shift(); // 수박 사과 

//splice ( 몇번째, 몇개를지울껀지)
fruits.push('딸기') //수박 사과 바나나 딸기
fruits.splice(1,1) //수박 바나나 
fruits.splice(1,1,'포도','귤'); //수박 포도 귤 
// fruits.splice(1) // 수박


//concat
const fruits2 = ['자몽','코코넛'];
const newFruits = fruits.concat(fruits2); //수박 포도 귤 자몽 

//searching
console.log(fruits.indexOf('포도')); // 1
console.log(fruits.includes('포도')); //true
console.log(fruits.indexOf('오렌지')); // -1   없을시-1

//lastIndexOf
console.log(fruits.lastIndexOf('포도')); 
```
